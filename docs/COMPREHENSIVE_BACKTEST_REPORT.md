## 🔧 技术实现

### 1. 市场状态识别

```python
def identify_market(stock_code):
    """识别市场状态"""
    df = load_data(stock_code)

    # 计算指标
    df['ma5'] = df['close'].rolling(5).mean()
    df['ma20'] = df['close'].rolling(20).mean()
    df['macd'], df['signal'], _ = macd(df['close'])
    df['rsi'] = rsi(df['close'])

    # 判断市场状态
    if df['ma5'].iloc[-1] > df['ma20'].iloc[-1] and df['macd'].iloc[-1] > 0:
        return 'BULL'  # 牛市
    elif 30 < df['rsi'].iloc[-1] < 70:
        return 'RANGE'  # 震荡
    else:
        return 'BEAR'  # 熊市
```

def check_stop_loss_profit(position, current_price):
    """检查止损止盈"""
    pnl_pct = (current_price - position['cost']) / position['cost']

    if pnl_pct < -0.08:  # 止损-8%
        return 'STOP_LOSS'
    elif pnl_pct > 0.20:  # 止盈+20%
        return 'TAKE_PROFIT'
    else:
        return 'HOLD'
